LOGIN
create table tb_usuario (
id_usuario int primary key auto_increment,
nome_usuario varchar (100),
senha_usuario varchar (100)
);


ENGLISH
create table tb_words (
id_word int primary key auto_increment,
word varchar (100),
translation varchar (100),
type enum (word, expression),
phrase varchar (100),
music_id int,
foreign key (music_id) references tb_musics(id_music),
data date,
id_usuario int primary key auto_increment,
foreign key (id_usuario) references tb_usuario(id_usuario)
);

create table tb_musics (
id_music int primary key auto_increment,
music_name varchar (100),
artist varchar (100),
length varchar (100),
lyric varchar (1000),
translation varchar (1000),
cover mediumblob,
music_link varchar (500),
visto boolean,
data date,
id_usuario int primary key auto_increment,
foreign key (id_usuario) references tb_usuario(id_usuario)
);


TREINO
create table tb_mes (
id_mes int primary key auto_increment,
mes varchar (100),
qtd_dias int,
id_usuario int primary key auto_increment,
foreign key (id_usuario) references tb_usuario(id_usuario)
);

create table tb_dia_concluido (
id_dia_concluido int primary key auto_increment,
dia varchar,
data date,
concluido boolean,
dificuldade enum ('Facil','Medio','Dificil'),
duracao varchar(100),
qtd_exercicios int,
dia_semana enum (seg, ter, qua, qui, sex, sab),
id_mes int,
id_usuario int primary key auto_increment,
foreign key (id_mes) references tb_mes(id_mes),
foreign key (id_usuario) references tb_usuario(id_usuario)
);

create table tb_exercicio (
id_exercicio int primary key auto_increment,
nome_exercicio varchar (100),
tipo enum (serie, tempo),
repeticoes int,
series int,
duracao int,
qtd_partes_corpo int,
dia_semana enum (seg, ter, qua, qui, sex, sab),
imagem mediumblob,
link_como_fazer varchar (500),
ordem int auto increment,
data date,
id_dia_concluido int,
id_usuario int primary key auto_increment,
foreign key (id_dia_concluido) references tb_dia_concluido(id_dia_concluido),
foreign key (id_usuario) references tb_usuario(id_usuario)
);

create table tb_partes_corpo (
id_parte_corpo int primary key auto_increment,
parte varchar(100), 
exercicio_id int,
foreign key (exercicio_id) references tb_exercicio(id_exercicio),
id_usuario int primary key auto_increment,
foreign key (id_usuario) references tb_usuario(id_usuario)
);

);

);

create table tb_serie (

create table tb_qtd_temporadas (

create table tb_eps_temporada (
